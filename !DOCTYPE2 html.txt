    const LS_KEY = 'archery_entries_v1';
    const todayStr = () => new Date().toISOString().slice(0,10);

    function loadEntries() {
      try {
        const raw = localStorage.getItem(LS_KEY);
        if (!raw) return [];
        const arr = JSON.parse(raw);
        return Array.isArray(arr) ? arr : [];
      } catch (e) { return []; }
    }
    function saveEntries(entries) {
      localStorage.setItem(LS_KEY, JSON.stringify(entries));
    }

    function markToChar(v){
      if (v === 1) return 'â—‹';
      if (v === -1) return 'Ã—';
      return '';
    }
    function nextMark(v){
      if (v === 0) return 1;   // ç©º â†’ â—‹
      if (v === 1) return -1;  // â—‹ â†’ Ã—
      return 0;                // Ã— â†’ ç©º
    }
    function countStats(marks){
      let o=0, x=0, e=0; // â—‹, Ã—, ç©º
      for (const m of marks){ if(m===1) o++; else if(m===-1) x++; else e++; }
      return {o,x,e};
    }

    function renderReport(){
      // æ—¥æœŸç¯„åœ
      const start = document.getElementById('range-start').value;
      const end = document.getElementById('range-end').value;
      const inRange = (d) => {
        if(start && d < start) return false;
        if(end && d > end) return false;
        return true;
      };

      const list = entries
        .map((e, i)=>({...e, rowIndex:i+1}))
        .filter(e => inRange(e.date))
        .sort((a,b) => a.date.localeCompare(b.date) || a.rowIndex - b.rowIndex);

      // è¨ˆç®—ï¼šç¸½ä¸­é¶ç‡ï¼ˆåªè¨ˆå…¥æœ‰æ¨™è¨˜çš„æ ¼ï¼šâ—‹ æˆ– Ã—ï¼‰
      const sum = {hits:0, tries:0};
      const arrow = [0,1,2,3].map(()=>({hits:0, tries:0})); // å‰å››ç®­
      list.forEach(e => {
        e.marks.forEach((m, idx) => {
          if(m!==0){ sum.tries++; if(m===1) sum.hits++; }
          if(idx < 4){ if(m!==0){ arrow[idx].tries++; if(m===1) arrow[idx].hits++; } }
        });
      });
      const pct = (h,t) => t===0 ? 'â€”' : (Math.round((h/t)*1000)/10).toFixed(1)+'%';
      // æ›´æ–°ç¸½ä¸­é¶ç‡
      document.getElementById('overall-rate').textContent = pct(sum.hits, sum.tries);
      document.getElementById('overall-sample').textContent = sum.tries ? `æ¨£æœ¬ï¼š${sum.hits}/${sum.tries}` : 'ç„¡è³‡æ–™';
      // æ›´æ–°å„ç®­ä¸­é¶ç‡

        const r = pct(arrow[i].hits, arrow[i].tries);
        document.getElementById(`a${n}-rate`).textContent = r;
        document.getElementById(`a${n}-sample`).textContent = arrow[i].tries ? `æ¨£æœ¬ï¼š${arrow[i].hits}/${arrow[i].tries}` : 'ç„¡è³‡æ–™';
      });

      // æ˜ç´°è¡¨
      reportBody.innerHTML = '';
      list.forEach(e => {
        const tr = document.createElement('tr');

        `;
        reportBody.appendChild(tr);
      });
    }
